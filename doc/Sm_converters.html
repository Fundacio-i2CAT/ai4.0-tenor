def bw_conversion(bw_string)
    bw_units = {
        "kbits" => 0.001,
        "mbits" => 1,
        "gbits" => 1000,
        "kbps"  => 0.001,
        "mbps"  => 1,
        "gbps"  => 1000,
    }
    bw_value = bw_string[/[0-9]+/]
    bw_unit  = bw_string[/\D+/]
    return bw_value.to_i * bw_units[bw_unit.downcase].to_i
end
